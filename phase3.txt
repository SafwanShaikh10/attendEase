1. Schema Fix — Add resubmission_requested_by (CRITICAL — DO FIRST)
This is the most important fix in Phase 3. The previous plan had no way to route a resubmitted request back to the correct approver level. This must be added to the schema BEFORE any controller code is written.
Add this field to the attendance_requests table:
-- Migration: add to attendance_requests
ALTER TABLE attendance_requests
  ADD COLUMN resubmission_requested_by
  ENUM('CLASS_COORD', 'YEAR_COORD', 'CHAIRPERSON') NULL;
Update the Prisma schema accordingly:
model attendance_requests {
  // ... existing fields ...
  resubmission_requested_by  ResubmissionRole?   // NEW
}
enum ResubmissionRole {
  CLASS_COORD
  YEAR_COORD
  CHAIRPERSON
}
How it is used — Resubmission routing logic:
When any approver calls POST /request-resubmission/:id, store their role in resubmission_requested_by.
When a student resubmits (creates a new request linked via parent_request_id), the system checks the parent's resubmission_requested_by.
The new request's starting status is set based on that field:
CLASS_COORD  →  new request starts at SUBMITTED  (goes to Class Coord first)
YEAR_COORD   →  new request starts at CLASS_COORD_APPROVED  (skips Class Coord)
CHAIRPERSON  →  new request starts at YEAR_COORD_APPROVED  (goes directly to Chairperson)
// In resubmit controller (student resubmits):
const parent = await prisma.attendance_requests.findUnique({
  where: { id: parentRequestId }
})
const startingStatus = {
  CLASS_COORD:  'SUBMITTED',
  YEAR_COORD:   'CLASS_COORD_APPROVED',
  CHAIRPERSON:  'YEAR_COORD_APPROVED'
}[parent.resubmission_requested_by]
await prisma.attendance_requests.create({
  data: {
    ...newRequestData,
    status: startingStatus,
    parent_request_id: parentRequestId
  }
})

Place 2 — Midnight cron job (auto-reject overdue SUBMITTED requests):
File: backend/jobs/deadline.job.js
const cron = require('node-cron')
const { PrismaClient } = require('@prisma/client')
const prisma = new PrismaClient()
// Runs every day at midnight
cron.schedule('0 0 * * *', async () => {
  const overdue = await prisma.attendance_requests.findMany({
    where: {
      status: 'SUBMITTED',
      deadline_date: { lt: new Date() }   // deadline_date = from_date + 6 days
    }
  })
  for (const req of overdue) {
    await prisma.attendance_requests.update({
      where: { id: req.id },
      data: {
        status: 'REJECTED',
        rejected_by: null,
        rejection_reason: 'Auto-rejected: 6-day deadline from absence date exceeded.'
      }
    })
    await prisma.audit_logs.create({
      data: {
        requestId: req.id,
        action: 'REJECTED',
        performedBy: null,
        performedAs: 'SYSTEM',
        note: 'Auto-rejected by deadline enforcement job (from_date + 6 days).'
      }
    })
  }
  console.log(`[CRON] Auto-rejected ${overdue.length} overdue requests.`)
})

const approve = async (req, res) => {
  const request = await prisma.attendance_requests.update({
    where: { id: Number(req.params.id) },
    data: { status: 'CHAIRPERSON_APPROVED' }  // Terminal status for SPECIAL_OD
  })
  // No balance deduction — Special OD is unlimited
  await excelService.appendRow(request)         // Same service built in Phase 2
  await googleDriveService.sync(request)        // Same service built in Phase 2
  await prisma.audit_logs.create({
    data: { requestId: request.id, action: 'CHAIRPERSON_APPROVED',
            performedBy: req.user.id, performedAs: 'CHAIRPERSON' }
  })
  // Notify student (Phase 4 notification service)
  res.json({ message: 'Special OD approved. Records updated.' })
}
POST /reject/:id:
const reject = async (req, res) => {
  const { reason } = req.body
  if (!reason) return res.status(400).json({ error: 'Rejection reason is mandatory.' })
  await prisma.attendance_requests.update({
    where: { id: Number(req.params.id) },
    data: { status: 'REJECTED', rejected_by: 'CHAIRPERSON', rejection_reason: reason }
  })
  await prisma.audit_logs.create({
    data: { requestId: Number(req.params.id), action: 'REJECTED',
            performedBy: req.user.id, performedAs: 'CHAIRPERSON', note: reason }
  })
  res.json({ message: 'Request rejected.' })
}
POST /request-resubmission/:id — stores which level triggered it:
const requestResubmission = async (req, res) => {
  const { note } = req.body
  if (!note) return res.status(400).json({ error: 'Resubmission note is mandatory.' })
  await prisma.attendance_requests.update({
    where: { id: Number(req.params.id) },
    data: {
      status: 'RESUBMISSION_REQUESTED',
      resubmission_note: note,
      resubmission_requested_by: 'CHAIRPERSON'   // NEW — critical for routing
    }
  })
  await prisma.audit_logs.create({
    data: { requestId: Number(req.params.id), action: 'RESUBMISSION_REQUESTED',
            performedBy: req.user.id, performedAs: 'CHAIRPERSON', note }
  })
  res.json({ message: 'Resubmission requested.' })
}

4.1  Answers to Open Questions
Q1 — letter_content or standard template?
Use the fixed template structure. The template headings, addressee, subject line, and sign-off are all fixed and non-editable. Only the student's reason field is injected dynamically into the body paragraph. Do not let the student freewrite the entire letter body.

4.2  Option A — Digital Letter (PDF Auto-Generation)
File: backend/services/letter.service.js
College name is Atria Institute of Technology — not any other institution.
const PDFDocument = require('pdfkit')
const fs = require('fs')
const path = require('path')
const LETTER_DIR = path.resolve('./uploads/letters')
if (!fs.existsSync(LETTER_DIR)) fs.mkdirSync(LETTER_DIR, { recursive: true })
const SUBJECT_LINES = {
  OD:         'Application for On-Duty (OD) Leave',
  MEDICAL:    'Application for Medical Leave',
  SPECIAL_OD: 'Application for Special On-Duty (Special OD) Leave'
}
const ADDRESSEES = {
  OD:         'The Class Coordinator',
  MEDICAL:    'The Class Coordinator',
  SPECIAL_OD: 'The Chairperson'          // Via Class Coord & Year Coord
}
async function generateLetterPDF(request, student) {
  return new Promise((resolve, reject) => {
    const fileName = `letter_${request.id}_${Date.now()}.pdf`
    const filePath = path.join(LETTER_DIR, fileName)
    const doc = new PDFDocument({ margin: 72 })
    const stream = fs.createWriteStream(filePath)
    doc.pipe(stream)
    // -- Fixed header (template) --
    doc.fontSize(14).font('Helvetica-Bold')
       .text('ATRIA INSTITUTE OF TECHNOLOGY', { align: 'center' })
    doc.fontSize(11).font('Helvetica')
       .text('Department of ' + student.department, { align: 'center' })
    doc.moveDown()
       .text('Bangalore', { align: 'center' })
    doc.moveDown(2)
    // -- Date (right-aligned) --
    doc.text('Date: ' + new Date().toDateString(), { align: 'right' })
    doc.moveDown()
    // -- Addressee (fixed per leave type) --
    const addressee = ADDRESSEES[request.leaveType]
    doc.text('To,')
    doc.text(addressee)
    if (request.leaveType === 'SPECIAL_OD') {
      doc.text('(Via: Class Coordinator & Year-wise Coordinator)')
    }
    doc.text('Atria Institute of Technology, Bangalore')
    doc.moveDown()
    // -- Subject (fixed per leave type) --
    doc.font('Helvetica-Bold')
       .text('Subject: ' + SUBJECT_LINES[request.leaveType])
    doc.font('Helvetica').moveDown()
    // -- Salutation --
    doc.text('Respected Sir/Madam,').moveDown(0.5)
    // -- Body (only reason is dynamic) --
    doc.text(
      `I, ${student.name}, Register No. ${student.registerNo || student.id}, ` +
      `a student of ${student.department} (Year ${student.year}, Division ${student.division}), ` +
      `humbly request your approval for leave from ` +
      `${new Date(request.fromDate).toDateString()} to ${new Date(request.toDate).toDateString()} ` +
      `(${request.daysCount} day(s)).`
    )
    doc.moveDown(0.5)
    doc.text('Reason: ' + request.reason)   // Only dynamic part
    doc.moveDown(0.5)
    doc.text('I assure you that I will make up for any missed academic work at the earliest.')
    doc.moveDown(2)
    // -- Sign-off (fixed) --
    doc.text('Yours sincerely,')
    doc.moveDown(0.5)
    doc.font('Helvetica-Bold').text(student.name)
    doc.font('Helvetica')
       .text('Register No: ' + (student.registerNo || student.id))
       .text('Year ' + student.year + ' | Division ' + student.division)
       .text('Department of ' + student.department)
    doc.end()
    stream.on('finish', () => resolve({ fileName, filePath }))
    stream.on('error', reject)
  })
}
module.exports = { generateLetterPDF }
Hook into submit controller (request.controller.js):
